/**
 * 前端应用主逻辑
 */

class OAuthAdminApp {
    constructor() {
        this.user = null;
        this.metrics = {};
        this.activeTab = 'dashboard';
        this.apiBase = '/api/v1';
        
        this.init();
    }

    init() {
        // 检查当前页面
        if (document.querySelector('.login-page')) {
            this.initLoginPage();
        } else {
            this.initMainApp();
        }
    }

    initLoginPage() {
        const loginForm = document.getElementById('loginForm');
        if (loginForm) {
            loginForm.addEventListener('submit', (e) => this.handleLogin(e));
        }

        // 检查 URL 参数
        const urlParams = new URLSearchParams(window.location.search);
        const error = urlParams.get('error');
        if (error) {
            this.showError(error);
        }
    }

    initMainApp() {
        // 初始化 Vue 应用或直接操作 DOM
        this.checkAuth();
        this.loadMetrics();
        this.setupEventListeners();
    }

    async checkAuth() {
        try {
            const response = await fetch(`${this.apiBase}/users/me`, {
                headers: {
                    'Authorization': `Bearer ${this.getToken()}`
                }
            });

            if (!response.ok) {
                window.location.href = '/static/login.html';
                return;
            }

            this.user = await response.json();
            this.updateUserInfo();
        } catch (error) {
            console.error('认证检查失败:', error);
            window.location.href = '/static/login.html';
        }
    }

    async handleLogin(event) {
        event.preventDefault();
        
        const formData = new FormData(event.target);
        const username = formData.get('username');
        const password = formData.get('password');

        try {
            const response = await fetch(`${this.apiBase}/oauth/token`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded',
                },
                body: new URLSearchParams({
                    grant_type: 'password',
                    client_id: 'admin_ui',
                    username: username,
                    password: password
                })
            });

            const data = await response.json();

            if (response.ok) {
                this.setToken(data.access_token);
                
                // 重定向到原页面或首页
                const urlParams = new URLSearchParams(window.location.search);
                const redirectUri = urlParams.get('redirect_uri') || '/static/index.html';
                window.location.href = redirectUri;
            } else {
                this.showError(data.error_description || '登录失败');
            }
        } catch (error) {
            console.error('登录请求失败:', error);
            this.showError('网络错误，请稍后重试');
        }
    }

    async loadMetrics() {
        try {
            const response = await fetch(`${this.apiBase}/admin/metrics`, {
                headers: {
                    'Authorization': `Bearer ${this.getToken()}`
                }
            });

            if (response.ok) {
                this.metrics = await response.json();
                this.updateMetricsDisplay();
            }
        } catch (error) {
            console.error('加载指标失败:', error);
        }
    }

    setActiveTab(tabName) {
        this.activeTab = tabName;
        this.updateTabDisplay();
        this.loadTabData(tabName);
    }

    async loadTabData(tabName) {
        switch (tabName) {
            case 'users':
                await this.loadUsers();
                break;
            case 'clients':
                await this.loadClients();
                break;
            case 'sync':
                await this.loadSyncStatus();
                break;
            case 'metrics':
                await this.loadMetrics();
                break;
        }
    }

    async loadUsers() {
        try {
            const response = await fetch(`${this.apiBase}/users?limit=50`, {
                headers: {
                    'Authorization': `Bearer ${this.getToken()}`
                }
            });

            if (response.ok) {
                const users = await response.json();
                this.renderUsersTable(users);
            }
        } catch (error) {
            console.error('加载用户列表失败:', error);
        }
    }

    async loadClients() {
        try {
            const response = await fetch(`${this.apiBase}/clients`, {
                headers: {
                    'Authorization': `Bearer ${this.getToken()}`
                }
            });

            if (response.ok) {
                const clients = await response.json();
                this.renderClientsTable(clients);
            }
        } catch (error) {
            console.error('加载客户端列表失败:', error);
        }
    }

    async loadSyncStatus() {
        try {
            const response = await fetch(`${this.apiBase}/admin/sync/status`, {
                headers: {
                    'Authorization': `Bearer ${this.getToken()}`
                }
            });

            if (response.ok) {
                const syncStatus = await response.json();
                this.renderSyncStatus(syncStatus);
            }
        } catch (error) {
            console.error('加载同步状态失败:', error);
        }
    }

    renderUsersTable(users) {
        // 实现用户表格渲染
        console.log('渲染用户表格:', users);
    }

    renderClientsTable(clients) {
        // 实现客户端表格渲染
        console.log('渲染客户端表格:', clients);
    }

    renderSyncStatus(syncStatus) {
        // 实现同步状态渲染
        console.log('渲染同步状态:', syncStatus);
    }

    updateUserInfo() {
        const userInfoElement = document.querySelector('.user-info span');
        if (userInfoElement && this.user) {
            userInfoElement.textContent = this.user.username;
        }
    }

    updateMetricsDisplay() {
        // 更新指标显示
        const statsGrid = document.querySelector('.stats-grid');
        if (statsGrid && this.metrics.user_metrics) {
            // 更新统计卡片
            console.log('更新指标显示:', this.metrics);
        }
    }

    updateTabDisplay() {
        // 更新标签页显示
        const tabs = document.querySelectorAll('.tab-content');
        tabs.forEach(tab => {
            tab.style.display = 'none';
        });
        
        const activeTab = document.querySelector(`[data-tab="${this.activeTab}"]`);
        if (activeTab) {
            activeTab.style.display = 'block';
        }
    }

    setupEventListeners() {
        // 设置侧边栏点击事件
        const sidebarLinks = document.querySelectorAll('.sidebar a');
        sidebarLinks.forEach(link => {
            link.addEventListener('click', (e) => {
                e.preventDefault();
                const tab = link.getAttribute('href').substring(1);
                this.setActiveTab(tab);
            });
        });

        // 退出按钮
        const logoutBtn = document.querySelector('.user-info button');
        if (logoutBtn) {
            logoutBtn.addEventListener('click', () => this.logout());
        }
    }

    logout() {
        this.removeToken();
        window.location.href = '/static/login.html';
    }

    getToken() {
        return localStorage.getItem('access_token');
    }

    setToken(token) {
        localStorage.setItem('access_token', token);
    }

    removeToken() {
        localStorage.removeItem('access_token');
    }

    showError(message) {
        const errorElement = document.getElementById('errorMessage');
        if (errorElement) {
            errorElement.textContent = message;
            errorElement.style.display = 'block';
        }
    }

    hideError() {
        const errorElement = document.getElementById('errorMessage');
        if (errorElement) {
            errorElement.style.display = 'none';
        }
    }
}

// 初始化应用
document.addEventListener('DOMContentLoaded', () => {

// API 工具函数
class ApiClient {
    constructor(baseUrl = '/api/v1') {
        this.baseUrl = baseUrl;
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseUrl}${endpoint}`;
        const token = localStorage.getItem('access_token');
        
        const config = {
            headers: {
                'Content-Type': 'application/json',
                ...options.headers
            },
            ...options
        };

        if (token) {
            config.headers.Authorization = `Bearer ${token}`;
        }

        try {
            const response = await fetch(url, config);
            
            if (response.status === 401) {
                // 未授权，跳转到登录页
                localStorage.removeItem('access_token');
                window.location.href = '/static/login.html';
                return;
            }

            if (!response.ok) {
                const error = await response.json().catch(() => ({}));
                throw new Error(error.error_description || `HTTP ${response.status}`);
            }

            return await response.json();
        } catch (error) {
            console.error('API 请求失败:', error);
            throw error;
        }
    }

    get(endpoint, options = {}) {
        return this.request(endpoint, { ...options, method: 'GET' });
    }

    post(endpoint, data = {}, options = {}) {
        return this.request(endpoint, {
            ...options,
            method: 'POST',
            body: JSON.stringify(data)
        });
    }

    put(endpoint, data = {}, options = {}) {
        return this.request(endpoint, {
            ...options,
            method: 'PUT',
            body: JSON.stringify(data)
        });
    }

    delete(endpoint, options = {}) {
        return this.request(endpoint, { ...options, method: 'DELETE' });
    }
}

// 创建全局 API 客户端
window.api = new ApiClient();
